
class file_reader {
  int sss;
  void close() {}
}

class token {
  string value;
  int f;
  token() {
    value = "x";
    f = 1;
  }
}

class type {
  token pos;
  string repr;
  file_reader fp;
  type() {
    pos = new token();
    repr = "int";
    fp = new file_reader();
  }
  deinit {
    fp.close();
  }
}

class opt {
  int sss;
  opt(){}
  void fn(string s) {
    std.print("%s\n", s);
  }
}

class main_class {
  int main() {
    type tp = new type();
    token tok = tp.pos;
    std.print("%s\n", tok.value);
    std.print("%d\n", tok.f);
    
    opt ooo= new opt();
    ooo.fn(tok.value);
    
    return tok.f;
  }
}
