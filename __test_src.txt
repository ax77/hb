
class file_reader {
  void close() {}
}

class token {
  string value;
  int f;
  token() {
    value = "x";
    f = 1;
  }
}

class type {
  token pos;
  string repr;
  file_reader fp;
  type() {
    pos = new token();
    repr = "int";
  }
  deinit {
    fp.close();
  }
}

class main_class {
  int main() {
    type tp = new type();
    token tok = tp.pos;
    std.print("%s\n", tok.value);
    std.print("%d\n", tok.f);
    return tok.f;
  }
}
