import std.list;

class main_class {
  int main() {
    list<string> list = new list<string>();
    list.push_back("2");
    list.push_front("1");

    list_iter<string> iter = new list_iter<string>(list);
    while (iter.has_next()) {
      string s = iter.get_next();
      std.print("%s\n", s);
    }
    return list.size();
  }
}
