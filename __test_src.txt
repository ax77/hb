
class opts {
  int i;
  opts() { i=0; }
}

class main_class {
  int main(int argc) {
    int i = 0;
    opts o = new opts();
    boolean b = true;
    
    static_assert( is_int(i) );
    static_assert( is_int(1) );
    static_assert( is_int(argc) );
    static_assert( is_char('1') );
    
    static_assert( is_reference(o) );
    static_assert( is_boolean(b) );
    
    return 0;
  }
}
