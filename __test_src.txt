
interface markable {
  void mark();
}

class str implements markable {
  boolean is_marked;
  str() {
    this.is_marked = false;
  }
  void mark() {
    this.is_marked = true;
  }
}

class ptr implements markable {
  boolean is_marked;
  ptr() {
    this.is_marked = false;
  }
  void mark() {
    this.is_marked = true;
  }
}

class main_class {
  int main() {
    markable s = new str();
    markable p = new ptr();
    s.mark();
    p.mark();
    return 0;
  }
}
