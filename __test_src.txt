class array<T> {
  native array();
  native void add(T e);
  native int size();
  native T get(int index);
  native T set(int index, T e);
}

class main_class {
  int main() {
    array<char> str = new array<char>();
    str.add('a');
    str.add('b');
    str.add('c');
    
    return str.size() == 3 && str.get(1) == 'b';
  }
}
