static inline void assert_true(
  int cnd, 
  struct string *file, 
  int line, 
  struct string *expr)  
{                                                                                      
    assert(file);
    assert(file->buffer);                                                                      
    assert(expr);
    assert(expr->buffer);                                                                      
    if (cnd == 0) {                                                                    
        fprintf(stdout, "assertion failed: (%s:%d) : [%s]\n"
          , file->buffer
          , line
          , expr->buffer);    
        exit(7);                                                                       
    }                                                                                  
}
